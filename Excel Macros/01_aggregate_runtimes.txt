Sub aggregate_runtimes()
'run the macro in the column to the immediate right
'of the disaggregated data

'determine if the system is running, if so, log a minute
Do Until IsEmpty(ActiveCell.Offset(0, -1).Value)
    If ActiveCell.Offset(0, -1) > 0.005 Then
        i = 1 + i
        ActiveCell.Offset(1, 0).Range("A1").Select
    Else
        ActiveCell.Offset(1, 0).Range("A1").Select
        i = 0
    End If

'determine when runtime cycle ends
'label total runtime (in minutes) next to beginning of cycle
'otherwise, paste 0
    If ActiveCell.Offset(0, -1).Value < .05 Then
        ActiveCell.Offset(-i, 0).Value = i
        ActiveCell.Value = 0
    Else
    End If
Loop
'when macro finishes, you can delete all rows with 0 or blank
'or run the capacity macro
End Sub
