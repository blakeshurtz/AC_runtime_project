Sub aggregate_capacity()
'run the macro in the column to the immediate right
'of the disaggregated data

'aggregate the data
Do Until IsEmpty(ActiveCell.Offset(0, -1).Value)
    If ActiveCell.Offset(0, -1) > 0.005 Then
        i = ActiveCell.Offset(0, -1).Value + i
        ActiveCell.Offset(1, 0).Range("A1").Select
        j = j + 1
    Else
        ActiveCell.Offset(1, 0).Range("A1").Select
    End If
    
'determine when runtime cycle ends
'for each runtime,
'label total capacity (in BTUs) in row containing the beginning of cycle
'otherwise, paste 0
    If ActiveCell.Offset(0, -1).Value < 0.05 Then
'output will be pasted one column over
        ActiveCell.Offset(-j, 1).Value = i
        ActiveCell.Offset(0, 1).Value = 0
        j = 0
        i = 0
    Else
    End If
Loop
'when macro finishes, you can delete all rows with 0 or blank
End Sub
